    ```typescript
    type Step = 'address' | 'confirm'
    const [step, setStep] = useState<Step>('address')
    const [deliveryAddress, setDeliveryAddress] = useState<'current' | 'mailing'>('current')
    const [mailingAddressText, setMailingAddressText] = useState('')
    const [passcodeOpen, setPasscodeOpen] = useState(false)
    const [isLoading, setIsLoading] = useState(false)
    const [resultModal, setResultModal] = useState<{
      type: 'success' | 'fail'

      error?: string
    } | null>(null)
    ```

    ```typescript
    const CARD_FEE_THB = 200
    const PROCESSING_FEE_THB = 10
    const TOTAL_THB = CARD_FEE_THB + PROCESSING_FEE_THB
    const EXCHANGE_RATE = 133.0
    const MMK_EQUIVALENT = Math.round(CARD_FEE_THB * EXCHANGE_RATE)  // 26600
    ```

    **Step 1 — Address Selection screen:**
    - BackHeader with t('card.request.title'), onBack → router.back()
    - Heading: t('card.request.subtitle')
    - Two radio-style cards (white bg, rounded-xl, border):
      - "Current Address" option: tapping sets deliveryAddress='current', shows green checkmark on selected
      - "Mailing Address" option: tapping sets deliveryAddress='mailing', expands a textarea/input below it for mailingAddressText
    - Bottom CTA button (yellow, rounded-full): t('card.request.confirmCta') — disabled if deliveryAddress='mailing' && mailingAddressText.trim()===''
    - On CTA tap: setStep('confirm')

    **Step 2 — Confirmation screen:**
    - BackHeader with t('card.request.confirmTitle'), onBack → setStep('address')
    - Summary card (white bg rounded-xl p-4):
      - Row: t('card.request.cardFee') / t('card.request.cardFeeValue')  (200 THB)
      - Divider

      - Divider
      - Row bold: t('card.request.total') / `${TOTAL_THB} THB`
      - Divider
      - Row: t('card.request.fxRate') / `1 THB = ${EXCHANGE_RATE} MMK`
      - Divider
      - Row: t('card.request.youReceive') / `${MMK_EQUIVALENT.toLocaleString()} MMK`
    - Delivery address row below summary card: show selected address label (or mailingAddressText if mailing)
    - Bottom CTA: t('card.request.confirmCta') — tapping opens PasscodeSheet

    **PasscodeSheet onVerified → handlePayment:**
    ```typescript
    const handlePayment = async () => {
      setIsLoading(true)
      try {
        const addressLine = deliveryAddress === 'current' ? 'Current Address' : mailingAddressText
        const res = await fetch('/api/mock-payment/visa-card', {
          method: 'POST',
          headers: { 'Content-Type': 'application/json' },
          body: JSON.stringify({
            delivery_address: deliveryAddress,
            address_line: addressLine,
            amount_satang: 20000,   // 200 THB
            fee_satang: 1000,       // 10 THB
          }),
        })
        const data = await res.json()
        if (data.success) {
          setResultModal({ type: 'success', cardReference: data.card_reference })
        } else {
          setResultModal({ type: 'fail', error: data.error })
        }
      } catch {
        setResultModal({ type: 'fail', error: t('card.request.failBody') })
      } finally {
        setIsLoading(false)
      }
    }
    ```

    **Result modals (AlertDialog controlled by resultModal state):**

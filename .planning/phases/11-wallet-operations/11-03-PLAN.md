    ### 1. Create `src/hooks/use-bank-accounts.ts`

    ```typescript
    import useSWR from 'swr'
    import type { BankAccount } from '@/types'

    interface BankAccountsData {
      bank_accounts: BankAccount[]
    }

    const fetcher = (url: string) =>
      fetch(url).then((r) => {
        if (!r.ok) throw new Error('fetch failed')
        return r.json()
      })

    export function useBankAccounts() {
      return useSWR<BankAccountsData>('/api/bank-accounts', fetcher, {
        dedupingInterval: 10_000,
      })
    }
    ```

    `handleDelete()`:
    ```typescript
    async function handleDelete() {
      if (!deletingId) return
      try {
        const res = await fetch(`/api/bank-accounts?id=${deletingId}`, { method: 'DELETE' })
        const data = await res.json()
        if (res.status === 409) {
          toast.error(data.error ?? 'Cannot delete — pending withdrawal exists.')
          setDeletingId(null)
          return
        }
        if (!res.ok) {
          toast.error('Failed to delete bank account.')
          setDeletingId(null)
          return
        }
        await mutate()
        toast.success('Bank account deleted.')
      } catch {
        toast.error('Connection error.')
      } finally {
        setDeletingId(null)
      }
    }
    ```

    export default function AddBankPage() {
      const router = useRouter()
      const [isSubmitting, setIsSubmitting] = useState(false)
      const form = useForm<AddBankInput>({
        resolver: zodResolver(addBankSchema),
        defaultValues: { bank_name: '', account_number: '', account_name: '' },
      })

      async function onSubmit(data: AddBankInput) {
        setIsSubmitting(true)
        try {
          const res = await fetch('/api/bank-accounts', {
            method: 'POST',
            headers: { 'Content-Type': 'application/json' },
            body: JSON.stringify(data),
          })
          const json = await res.json()
          if (!res.ok) {
            toast.error(json.error ?? 'Failed to save bank account.')
            return
          }
          toast.success('Bank account saved.')
          router.push('/withdraw')
        } catch {
          toast.error('Connection error. Please try again.')
        } finally {
          setIsSubmitting(false)
        }
      }
      // ... render form
    }
